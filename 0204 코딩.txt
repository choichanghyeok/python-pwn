1. 나누어 떨어지는 숫자 배열

def solution(arr, divisor):
    answer = []
    count = 0
    for i in range(len(arr)):
        if arr[i]%divisor ==0:
            answer.append(arr[i])
            
        else:				# if not answer:
            count +=1				answer = [=1] 도 됨
            
        if count == len(arr):
            answer.append(-1)
            
    answer.sort()
    
    return answer

4. 문자열 내 p와 y의 개수

def solution(s):
    
    c = s.lower()
    cx = c.count('p')
    cy = c.count('y')
    
    if cx == cy:
        return True
    else:
        return False

6.이상한 문자 만들기



def solution(s):
    result = []
    
    for char in s.split(' '):
        idx  = 0
        for i in char:
            if idx %2 ==0:              # 총 3개의 조각 한개 한개 한개 씩 대소문자 다 싹다 바꿔줌 비교해서
                result.append(i.upper())    # idx 로 1씩증가로인해 char 은 3  관련 글자들 %2로 홀짝이 갈라지면서
            else:                           # 대소문자로 바뀜
                result.append(i.lower())
            idx +=1                         # 다음 글자 단어로 넘어가기위한 인덱스  0은 try , 1은 hello  2는 world
            
        result.append("*")                  #"T,r,Y,H,e,L,l,O,W,o,R,l,D  글자를 완성하면 이렇게 되어있기떄문에 띄어쓰기 넣어줌
                                            # 끝날떄마다 넣어주는거임
        
    result.pop()                            # 빈여백 삭제
    
    return "".join(result)                  # ''.join(result) 으로 result에있는 요소들을 전부 합쳐버림
