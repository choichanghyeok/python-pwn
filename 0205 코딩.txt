2. 소수 찾기

(이해 아직 조금밖에 안됬음 나중에 다시 이해해야함)

def solution(n):
    
    num = set(range(2,n+1))
    for i in range(2,n+1):
        if i in num:
            num -= set([i for i in range(i*2,n+1,i)])
        return len(num)+1

#set으로 중복제거 i * 3에서 시작 n+1까지 2씩증가

3. 서울에서 김서방 찾기

내풀이

def solution(seoul):
    
    for i in range(len(seoul)):
        if seoul[i] == "Kim":
            return "김서방은 "+str(i)+"에 있다"

다른 사람 풀이

def findKim(seoul):
    return "김서방은 {}에 있다".format(seoul.index('Kim'))
