1. 타겟 넘버



def solution(numbers, target):
    
    tar = [0]   # 첫 시작 (트리 기준으로 생각하면 쉬움)
    for i in numbers:
        num = []    # 해당 경로 또는 해당 숫자 더하거나 빼서 원하는 숫자가 나온 경우
        for j in tar:
            num.append(j-i)                 # 전체의 경우의 수를 구한다
            num.append(j+i)
        tar = num
    return tar.count(target)                # 전체에서 target만 개수를 세서 리턴한다.
        






2. 영어 끝말잇기



def solution(n, words):
    
    # 처음으로 말한 단어 저장
    
    first = set([words[0]])		# 여기에 먼저 [0]을 저장하기 떄문에 반복문돌릴떄 1부터시작
    #print(first)
    
    # 말했던 이전 단어 저장
    
    mid = words[0]

    
    for i in range(1,len(words)):
        # 이전 단어의 마지막 글자와 제시한 단어의 첫글자가 다르거나 , 이미 이는 단어일 경우
        
        if (mid[-1] != words[i][0]) or words[i] in first:
            
            #사람,차례  i == 사람( 몇 번째)   n == 차례
            return [(i%n)+1, (i//n)+1]			# 위에서 word[0]으로 하나가 빠졌기때문에 차례나 몇번쨰인지 알기위해서0 +1씩 해준다.
        
        # 말한단어 저장
        first.add(words[i])
        
        # 또 말한단어를 변수에 저장
        
        mid = words[i]
    return [0,0]
